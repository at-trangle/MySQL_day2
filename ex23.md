23. Lấy 2 blog, 2 news mà user có id = 1 đã comment
```
(
  select blog.id,blog.title,userComment.target_table
  from blog inner join (select target_id,target_table
  from comment
  where target_table='blog' AND user_id = 1
  limit 2) as userComment on userComment.target_id = blog.id
)
union all
(
  select news.id,news.title,userComment.target_table
  from news inner join (select target_id,target_table
  from comment
  where target_table='news' AND user_id = 1
  limit 2) as userComment on userComment.target_id = news.id
);
```
